\section{Introduction}
%==============================================================================

\subsection{The Pebas Mega-Wetland}

The Pebas Formation records one of the most remarkable freshwater ecosystems in Earth history---a mega-wetland exceeding 1 million km$^2$ that persisted for nearly 15 million years in western Amazonia.

\begin{itemize}
    \item \textbf{Endemic mollusc radiations}: 158 species, 72\% endemic
    \item \textbf{Exceptional vertebrate preservation}: articulated crocodilians, giant turtles, freshwater dolphins
    \item \textbf{Time period}: Our cores span $\sim$12.9--14.3 Ma (Middle Miocene)
    \item \textbf{Mollusc zones}: MZ7--MZ8 (peak biodiversity)
\end{itemize}

\subsection{Research Questions}

\begin{enumerate}
    \item Can XRF proxies detect the dysoxic conditions documented by mollusc ecology?
    \item Do geochemical signals match independent isotopic evidence from bivalve shells?
    \item What drove environmental variability in the Pebas system?
    \item How do site differences reflect spatial heterogeneity vs.\ temporal evolution?
\end{enumerate}

\subsection{Proxy Rationale}

\begin{table}[H]
\centering
\caption{XRF proxy framework for paleoenvironmental reconstruction}
\begin{tabular}{lllcc}
\toprule
Proxy & Numerator & Denominator & Independent? & Interpretation \\
\midrule
\textbf{Mn/Ti} & Authigenic Mn oxides & Detrital Ti & $\checkmark$ Primary & High = oxic \\
\textbf{Ca/Ti} & Biogenic carbonate & Detrital Ti & $\checkmark$ Yes & High = shells \\
\textbf{Fe/Mn} & Mixed Fe sources & Authigenic Mn & $\times$ Shares Mn & High = reducing \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Mn redox chemistry}: Mn$^{4+}$ (insoluble) precipitates under oxic conditions as MnO$_2$. Under reducing conditions, Mn$^{4+}$ $\rightarrow$ Mn$^{2+}$ (soluble), which diffuses away. Threshold: $\sim$200 mV (Eh), $\sim$0.2 ml/L O$_2$.

\subsection{Temporal Caveat}

\begin{tcolorbox}[colback=red!5, colframe=red!50!black, title=Important Interpretive Note]
The site contrast (TAM dysoxic, SC oxic) may partly reflect:
\begin{itemize}[noitemsep]
    \item \textbf{Spatial heterogeneity}: TAM = distal lacustrine; SC = proximal fluvio-lacustrine
    \item \textbf{Temporal evolution}: SC records an older, perhaps more dynamic phase; TAM records a younger, more stable dysoxic phase
\end{itemize}
For time-equivalent comparison, focus on:
\begin{itemize}[noitemsep]
    \item TAM: bottom 133 cm (oldest TAM material)
    \item SC: top 71 cm (youngest SC material)
\end{itemize}
\end{tcolorbox}

\section{Case Studies: Paleoenvironmental Mechanisms}
%==============================================================================

\subsection{Case Study 1: Dramatic Oxygenation Event}

\textbf{Location}: SC-1ABC-2-3C-A-RUN1, depth 367 mm

\textbf{Signal}: +833\% increase in Mn/Ti over a few millimeters

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{../output/figures/temporal_overlap/fig3_case_oxygenation.png}
    \caption{Case Study 1: The most dramatic oxygenation event in the dataset. The +833\% Mn/Ti transition at 367 mm represents an abrupt ventilation event---the environment flipped from reducing to oxidizing conditions within years to decades.}
    \label{fig:case1}
\end{figure}

\textbf{Mechanism}: Three possibilities:
\begin{enumerate}
    \item \textbf{Andean runoff pulse}: Cold, oxygen-rich river water plunges into stratified lake
    \item \textbf{Lake-level drop}: Shallowing enables wind mixing to reach bottom
    \item \textbf{Storm mixing}: Episodic breakdown of thermal stratification
\end{enumerate}

\textbf{Preservation implication}: This oxygenation event would have temporarily disrupted the preservation-favorable dysoxic conditions, potentially causing bioturbation and scavenging of any carcasses present.

\subsection{Case Study 2: Persistent Dysoxic Baseline}

\textbf{Location}: TAM-1-2-3B-A, depth 243--444 mm

\textbf{Signal}: Sustained extremely low Mn/Ti (mean = 0.078), Fe/Mn > 400

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{../output/figures/temporal_overlap/fig4_case_dysoxia.png}
    \caption{Case Study 2: A 200 mm interval of extreme dysoxia at TAM-1-2-3B-A. Mn/Ti values as low as 0.078 indicate bottom water oxygen $<$0.2 ml/L---\textit{Pachydon obliquus} territory.}
    \label{fig:case2}
\end{figure}

\textbf{Mechanism}:
\begin{itemize}
    \item Thermal stratification prevents mixing with oxygenated surface water
    \item Organic matter accumulation consumes oxygen via bacterial respiration
    \item Water residence time high enough for O$_2$ depletion
\end{itemize}

\textbf{Preservation implication}: This is the ``taphonomic sweet spot.'' Conditions favor:
\begin{itemize}
    \item No scavengers (crocs, catfish can't survive)
    \item No bioturbation (infaunal organisms excluded except Pachydon)
    \item Organic matter preservation (no oxidative decay)
    \item Rapid burial in fine-grained organic-rich mud
\end{itemize}

\subsection{Case Study 3: Concordant Oxic-Carbonate Event}

\textbf{Location}: SC-3AB-4ABCD-RUN2-F, depth 1258 mm

\textbf{Signal}: Mn/Ti +5.8$\sigma$, Ca/Ti +2.0$\sigma$ (both moving together)

\begin{figure}[H]
    \centering
    \includegraphics[width=0.9\textwidth]{../output/figures/temporal_overlap/fig5_case_concordant.png}
    \caption{Case Study 3: Concordant oxic-carbonate event. When Mn/Ti and Ca/Ti move together, it's strong evidence for a real environmental event---not a compositional artifact. This interval represents oxygenation that simultaneously favored Mn precipitation and mollusc colonization.}
    \label{fig:case3}
\end{figure}

\textbf{Mechanism}:
\begin{itemize}
    \item Oxygenation event precipitates Mn as MnO$_2$ (increases Mn/Ti)
    \item Improved conditions allow mollusc colonization and shell production (increases Ca/Ti)
    \item Independent proxies responding to the same environmental driver
\end{itemize}

\textbf{Why this matters}: Mn and Ca have different source terms (authigenic vs.\ biogenic). Concordant anomalies resist the compositional data closure critique because they require a common environmental driver.

\section{Synthesis and Conclusions}
%==============================================================================

\subsection{Key Findings}

\begin{enumerate}
    \item \textbf{XRF detects dysoxia}: Mn/Ti successfully identifies reducing conditions independently documented by \textit{Pachydon obliquus} ecology (Wesselingh 2006).

    \item \textbf{Site contrast = spatial + temporal}: TAM (0.18 Mn/Ti) vs SC (0.37 Mn/Ti) reflects both environmental heterogeneity and temporal evolution. Only 171 kyr of overlap.

    \item \textbf{Concordant proxies = robust signals}: When Mn/Ti and Ca/Ti move together, the signal resists compositional closure critique.

    \item \textbf{Dynamic system}: 833\% Mn/Ti transitions record discrete ventilation events in an otherwise stratified system.

    \item \textbf{Preservation connection}: The same geochemistry that tracks paleoenvironment explains why Pebas preserves fossils exceptionally---dysoxia is a taphonomic factory.
\end{enumerate}

\subsection{Future Directions}

\begin{itemize}
    \item \textbf{Time-series analysis}: Test for orbital periodicities in Mn/Ti
    \item \textbf{Overlap comparison}: Focus on temporally equivalent intervals (TAM bottom vs.\ SC top)
    \item \textbf{Vertebrate taphonomy}: Use XRF to predict preservation quality before excavation
    \item \textbf{Regional correlation}: Link local XRF signals to basin-wide Pebas evolution
\end{itemize}
